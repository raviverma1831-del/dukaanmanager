4.3 Message Processing Logic
async function processMessage(message, shopId) {
  // 1. Get shop full context
  const context = await getShopContext(shopId);
  // 2. Detect message type
  if (message.type === 'audio') {
    message.text = await whisper.transcribe(message.audio);
  }
  if (message.type === 'image') {
    message.text = await claude.vision(message.image);
  }
  // 3. Send to Claude with full context
  const action = await claude.decide({
    userMessage: message.text,
    shopData: context,
    systemPrompt: DUKAANMANAGER_PROMPT
  });
  // 4. Execute action on database
  await executeAction(action, shopId);
  // 5. Send WhatsApp response
  await whatsapp.send(action.response, shopId);
}
